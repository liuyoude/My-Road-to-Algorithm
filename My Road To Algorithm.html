53.最大子序列和

给定一个整数数组 nums ，找到一个具有最大和的连续子数组（子数组最少包含一个元素），返回其最大和。
示例:
输入: [-2,1,-3,4,-1,2,1,-5,4]
输出: 6
解释: 连续子数组 [4,-1,2,1] 的和最大，为 6。
进阶:
如果你已经实现复杂度为 O(n) 的解法，尝试使用更为精妙的分治法求解。

解法1：贪心

如果之前序列和小于0就放弃之前序列

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int max_subsq_sum = nums[0], current_sum=0;
        for(int i = 0; i < nums.size(); i ++ ){
            // 如果之前序列和小于等于0，放弃之前序列，将当前值设为新序列和
            if(current_sum <= 0){
                current_sum = nums[i];
            }else{
            // 否则，将当前值和序列和相加作为新序列和
                current_sum += nums[i];
            }
            // 如果新序列和的值大于最大序列值，将最大子序列值设为当前值
            if(current_sum > max_subsq_sum){
                max_subsq_sum = current_sum;
            }
        }
        return max_subsq_sum;
    }
};
322.零钱兑换

给定不同面额的硬币 coins 和一个总金额 amount。编写一个函数来计算可以凑成总金额所需的最少的硬币个数。如果没有任何一种硬币组合能组成总金额，返回 -1。
你可以认为每种硬币的数量是无限的。
示例 1：
输入：coins = [1, 2, 5], amount = 11
输出：3
解释：11 = 5 + 5 + 1
提示：
1 <= coins.length <= 12
1 <= coins[i] <= 231 - 1
0 <= amount <= 104

    int solve(vector<int>& memo, vector<int>& coins, int amount){
        if(amount == 0) return 0;
        if(amount < 0 ) return -1;
        if(memo[amount] != 0) return memo[amount];
        int res = float('INF');
        for(int i = 0; i < coins.size(); i ++){
            int subres = solve(memo, coins, amount-coins[i]);
            if(subres == -1) continue;
            if(subres+1 < res){
                res = subres+1;
            }
        }
        if(res == float('INF')){
            return -1;
        }
        memo[amount] = res;
        return res;
    }

        dp[0] = 0;
        for(int i = 1; i <= amount; i ++ ){

            return -1;

509.菲波那切数列

斐波那契数，通常用 F(n) 表示，形成的序列称为斐波那契数列。该数列由 0 和 1 开始，后面的每一项数字都是前面两项数字的和。也就是：
F(0) = 0,   F(1) = 1
F(N) = F(N - 1) + F(N - 2), 其中 N > 1.
给定 N，计算 F(N)。
示例 1：
输入：2
输出：1
解释：F(2) = F(1) + F(0) = 1 + 0 = 1.
提示：
0 ≤ N ≤ 30

解法1.自底向上 DP (动态规划思想)
class Solution {
public:
    int fib(int N) {
        if(N <= 1) return N;
        vector<int> dp(N+1, 0);
        dp[1] = 1;
        // 记录之前所有状态值
        for(int i = 2; i <= N; i ++ ){
            dp[i] = dp[i-1] + dp[i-2];
        }
        return dp[N];
    }
    
    // 不需要存储所有状态，只需要存储前两个状态即可
    int fib_mini(int N) {
            if(N <= 1) return N;
            int pre = 0, cur = 1;
            for(int i = 2; i <= N; i ++ ){
                int sum = pre + cur;
                pre = cur;
                cur = sum;
            }
            return cur;
        }
};
解法2.自上向下 带备忘录的递归

可以直接通过递归解决，但这个过程中存在着大量重叠子问题，因此用备忘录存储重叠的子问题防止重复计算

    int fib(int N) {
            vector<int> memo(N+1, 0);
            return helper(memo, N);
        }
    int helper(vector<int>& memo, int N){
        if(N == 0) return 0;
        if(N == 1) return 1;
        // 如果备忘录中存储， 就直接返回备忘录的值
        if(memo[N] != 0) return memo[N];
        // 否则记录在备忘录中
        memo[N] = helper(memo, N-1) + helper(memo, N-2);
        return memo[N];
    }
